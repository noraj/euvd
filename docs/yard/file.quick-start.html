Quick start

Quick install

gem install euvd

Default usage: library

require 'euvd'

# Initialize client
client = EUVD::Client.new

# Search for a vuln
record = client.records.find('EUVD-2025-20101')

# Access attributes
record.description
record.baseScoreVector
